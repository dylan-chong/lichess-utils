Settings Store with Signals

settings/settingsStore.ts
import { signal, effect } from '@preact/signals-core'

export interface Settings {
  speakRate: number
  piecesListEnabled: boolean
  dividersEnabled: boolean
  customBoardEnabled: boolean
  obfuscationsEnabled: boolean
  parallax: number
  hoverMode: string
  pieceStyle: string
  blur: number
  blackSegments: string
  blackSegmentsTiming: string
  flashModeEnabled: boolean
  flashDuration: number
  flashInterval: number
}

// Each setting is its own signal
export const settings = {
  speakRate: signal(0.5),
  piecesListEnabled: signal(false),
  dividersEnabled: signal(false),
  customBoardEnabled: signal(false),
  obfuscationsEnabled: signal(false),
  parallax: signal(0),
  hoverMode: signal('off'),
  pieceStyle: signal('icons'),
  blur: signal(0),
  blackSegments: signal('none'),
  blackSegmentsTiming: signal('rotate-10s'),
  flashModeEnabled: signal(false),
  flashDuration: signal(1),
  flashInterval: signal(3),
}

const STORAGE_KEY = 'lichess-board-speaker-settings'

export function loadSettings() {
  const stored = localStorage.getItem(STORAGE_KEY)
  if (stored) {
    const data = JSON.parse(stored)
    Object.keys(data).forEach(key => {
      if (settings[key]) {
        settings[key].value = data[key]
      }
    })
  }
}

export function saveSettings() {
  const data = {}
  Object.keys(settings).forEach(key => {
    data[key] = settings[key].value
  })
  localStorage.setItem(STORAGE_KEY, JSON.stringify(data))
}

// Auto-save on any setting change
effect(() => {
  // Read all signals to track dependencies
  Object.values(settings).forEach(s => s.value)
  saveSettings()
})

import { h } from 'preact'
import { settings } from '../settings/settingsStore'
import { ButtonRow } from './ButtonRow'
import { PiecesList } from './PiecesList'

export function ControlPanel() {
  return (
    <div class="userscript-controls">
      {/* Main controls - always visible */}
      <ButtonRow buttons={mainButtons} />

      {/* Custom board controls - conditional */}
      {settings.customBoardEnabled.value && (
        <ButtonRow buttons={customBoardButtons} />
      )}

      {/* Obfuscation controls - nested conditional */}
      {settings.customBoardEnabled.value && settings.obfuscationsEnabled.value && (
        <ButtonRow buttons={obfuscationButtons} />
      )}

      {/* Black segments timing - nested conditional */}
      {settings.customBoardEnabled.value &&
       settings.obfuscationsEnabled.value &&
       settings.blackSegments.value !== 'none' && (
        <ButtonRow buttons={blackSegmentsTimingButtons} />
      )}

      {/* Flash mode controls - conditional */}
      {settings.flashModeEnabled.value && (
        <ButtonRow buttons={flashModeButtons} />
      )}

      {/* Pieces list - conditional */}
      {settings.piecesListEnabled.value && <PiecesList />}
    </div>
  )
}

components/SettingButton.tsx
import { h } from 'preact'
import { Signal } from '@preact/signals-core'

interface SettingButtonProps {
  label: string
  signal: Signal<any>
  options: any[]
  format?: (value: any) => string
}

export function SettingButton({ label, signal, options, format }: SettingButtonProps) {
  const handleClick = () => {
    const currentIndex = options.indexOf(signal.value)
    const nextIndex = (currentIndex + 1) % options.length
    signal.value = options[nextIndex]
  }

  const displayValue = format ? format(signal.value) : signal.value

  return (
    <button onClick={handleClick} class="setting-button">
      {label}: {displayValue}
    </button>
  )
}
components/PiecesList.tsx
import { h } from 'preact'
import { useSignal } from '@preact/signals'
import { useEffect } from 'preact/hooks'
import { boardWatcher } from '../board/BoardWatcher'
import { boardReader } from '../board/BoardReader'

export function PiecesList() {
  const pieces = useSignal([])

  const updatePieces = () => {
    pieces.value = boardReader.readPiecePositions()
  }

  useEffect(() => {
    updatePieces() // Initial read
    const cleanup = boardWatcher.onChange(updatePieces)
    return cleanup
  }, [])

  const grouped = groupPiecesByColorAndType(pieces.value)
  const playerColor = boardReader.getPlayerColor()

  return (
    <div class="pieces-list-box">
      <div>You are: {playerColor}</div>
      <div>Turn: {getCurrentTurn()}</div>

      {Object.entries(grouped).map(([color, piecesByType]) => (
        <div key={color}>
          <strong>{color}:</strong>
          {Object.entries(piecesByType).map(([type, squares]) => (
            <div key={type}>
              {type}: {squares.join(', ')}
            </div>
          ))}
        </div>
      ))}
    </div>
  )
}

Board Watcher (Simplified)

board/BoardWatcher.ts
export class BoardWatcher {
  private observer: MutationObserver | null = null
  private listeners: Array<() => void> = []

  onChange(callback: () => void): () => void {
    this.listeners.push(callback)
    return () => {
      const index = this.listeners.indexOf(callback)
      if (index > -1) this.listeners.splice(index, 1)
    }
  }

  private notifyListeners() {
    this.listeners.forEach(cb => cb())
  }

  start() {
    const board = document.querySelector('cg-board')
    if (!board) return

    this.observer = new MutationObserver(() => {
      this.notifyListeners()
    })

    this.observer.observe(board, {
      childList: true,
      attributes: true,
      subtree: true,
    })
  }

  stop() {
    this.observer?.disconnect()
    this.observers = null
  }
}

export const boardWatcher = new BoardWatcher()

Key patterns:

  • Settings are signals - components auto-update when values change
  • Conditional rendering with {signal.value && <Component />}
  • Button cycles through options array, updates signal directly
  • PiecesList uses useEffect to subscribe to board changes
  • BoardWatcher still uses callbacks (not everything needs signals)
