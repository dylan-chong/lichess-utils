Managers as Factory Functions

No classes, just closures and returned functions

BoardManager

board/boardManager.ts
import { effect } from '@preact/signals-core'
import { settings } from '../settings/settingsStore'
import { readPiecePositions } from './boardReader'
import { createBoardWatcher } from './boardWatcher'
import { createCustom3DBoard } from './custom3DBoard'
import { createDividers } from './dividers'

export function createBoardManager() {
  const watcher = createBoardWatcher()
  const dividers = createDividers()
  let board3D = null
  let cleanupEffect = null
  let cleanupWatcher = null

  function init() {
    watcher.start()

    // React to board changes
    cleanupWatcher = watcher.onChange(() => {
      if (board3D) {
        const pieces = readPiecePositions()
        board3D.render(pieces)
      }
    })

    // React to settings changes
    cleanupEffect = effect(() => {
      reapplyAll()
    })
  }

  function reapplyAll() {
    // Custom board
    if (settings.customBoardEnabled.value) {
      if (!board3D) {
        board3D = createCustom3DBoard()
        board3D.init()
      }
      board3D.updateParallax(settings.parallax.value)
      board3D.updateHoverMode(settings.hoverMode.value)

      if (settings.obfuscationsEnabled.value) {
        board3D.updatePieceStyle(settings.pieceStyle.value)
        board3D.updateBlur(settings.blur.value)
        board3D.updateBlackSegments(
          settings.blackSegments.value,
          settings.blackSegmentsTiming.value
        )
      }
    } else if (board3D) {
      board3D.destroy()
      board3D = null
    }

    // Dividers
    if (settings.dividersEnabled.value) {
      dividers.render(settings.customBoardEnabled.value)
    } else {
      dividers.hide()
    }
  }

  function destroy() {
    watcher.stop()
    board3D?.destroy()
    dividers.destroy()
    cleanupEffect?.()
    cleanupWatcher?.()
  }

  return { init, destroy, watcher }
}

FlashManager

flash/flashManager.ts
import { effect } from '@preact/signals-core'
import { settings } from '../settings/settingsStore'
import { createFlashOverlay } from './flashOverlay'

export function createFlashManager(boardWatcher) {
  const overlay = createFlashOverlay()
  let durationTimer = null
  let intervalTimer = null
  let cleanupEffect = null
  let cleanupWatcher = null

  function init() {
    // React to board changes
    cleanupWatcher = boardWatcher.onChange(() => {
      if (settings.flashModeEnabled.value) {
        onBoardChange()
      }
    })

    // React to settings changes
    cleanupEffect = effect(() => {
      reapply()
    })
  }

  function reapply() {
    if (settings.flashModeEnabled.value) {
      overlay.show()
      startIntervalFlash()
    } else {
      overlay.hide()
      stopIntervalFlash()
    }
  }

  function onBoardChange() {
    overlay.hide()

    if (durationTimer) clearTimeout(durationTimer)

    const duration = settings.flashDuration.value * 1000
    durationTimer = setTimeout(() => {
      overlay.show()
    }, duration)

    stopIntervalFlash()
    startIntervalFlash()
  }

  function startIntervalFlash() {
    const interval = settings.flashInterval.value * 1000
    intervalTimer = setInterval(onBoardChange, interval)
  }

  function stopIntervalFlash() {
    if (intervalTimer) {
      clearInterval(intervalTimer)
      intervalTimer = null
    }
  }

  function destroy() {
    if (durationTimer) clearTimeout(durationTimer)
    stopIntervalFlash()
    overlay.destroy()
    cleanupEffect?.()
    cleanupWatcher?.()
  }

  return { init, destroy }
}

SpeechManager

speech/speechManager.ts
import { effect } from '@preact/signals-core'
import { settings } from '../settings/settingsStore'
import { createSpeechSynthesizer } from './speechSynthesizer'
import { readPiecePositions } from '../board/boardReader'
import * as textGen from './textGenerators'

export function createSpeechManager() {
  const synthesizer = createSpeechSynthesizer()
  let cleanupEffect = null

  function init() {
    cleanupEffect = effect(() => {
      synthesizer.setRate(settings.speakRate.value)
    })
  }

  function speakQuadrant(quadrant) {
    const pieces = readPiecePositions()
    const text = textGen.generateQuadrantText(pieces, quadrant)
    synthesizer.speak(text)
  }

  function speakAll() {
    const pieces = readPiecePositions()
    const text = textGen.generateAllPiecesText(pieces)
    synthesizer.speak(text)
  }

  function speakColor(color) {
    const pieces = readPiecePositions()
    const text = textGen.generateColorText(pieces, color)
    synthesizer.speak(text)
  }

  function stop() {
    synthesizer.stop()
  }

  function destroy() {
    cleanupEffect?.()
  }

  return { init, speakQuadrant, speakAll, speakColor, stop, destroy }
}

DrawingManager

drawing/drawingManager.ts
import { parseDrawCommand } from './commandParser'
import { createDrawingsOverlay } from './drawingsOverlay'

export function createDrawingManager() {
  const overlay = createDrawingsOverlay()

  function draw(commandStr) {
    const commands = parseDrawCommand(commandStr)
    overlay.clear()
    overlay.draw(commands)
  }

  function clear() {
    overlay.clear()
  }

  function destroy() {
    overlay.destroy()
  }

  return { draw, clear, destroy }
}

Main Entry (Revised)

main.ts
import { render } from 'preact'
import { loadSettings } from './settings/settingsStore'
import { ControlPanel } from './components/ControlPanel'
import { createBoardManager } from './board/boardManager'
import { createSpeechManager } from './speech/speechManager'
import { createDrawingManager } from './drawing/drawingManager'
import { createFlashManager } from './flash/flashManager'
import { setupCommandHandler } from './commands/commandHandler'

async function waitForElement(selector) {
  return new Promise(resolve => {
    if (document.querySelector(selector)) {
      return resolve(document.querySelector(selector))
    }
    const observer = new MutationObserver(() => {
      if (document.querySelector(selector)) {
        resolve(document.querySelector(selector))
        observer.disconnect()
      }
    })
    observer.observe(document.body, { childList: true, subtree: true })
  })
}

async function init() {
  await waitForElement('.keyboard-move')

  loadSettings()

  const boardManager = createBoardManager()
  const speechManager = createSpeechManager()
  const drawingManager = createDrawingManager()
  const flashManager = createFlashManager(boardManager.watcher)

  boardManager.init()
  speechManager.init()
  flashManager.init()

  setupCommandHandler(speechManager, drawingManager)

  const mountPoint = document.createElement('div')
  document.querySelector('.keyboard-move')?.appendChild(mountPoint)
  render(<ControlPanel />, mountPoint)

  return () => {
    boardManager.destroy()
    speechManager.destroy()
    drawingManager.destroy()
    flashManager.destroy()
    render(null, mountPoint)
  }
}

init()

Factory function pattern:

  • Functions that return objects with methods
  • State lives in closure scope (private by default)
  • No this keyword, no new, no class
  • Simpler to test (just call the function, get back an object)
  • Dependencies passed as parameters (easier to see what depends on what)
