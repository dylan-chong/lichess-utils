Managers with Signal Effects

How board/speech/flash managers react to settings changes

BoardManager with Signals

board/BoardManager.ts
import { effect } from '@preact/signals-core'
import { settings } from '../settings/settingsStore'
import { BoardReader } from './BoardReader'
import { BoardWatcher } from './BoardWatcher'
import { Custom3DBoard } from './Custom3DBoard'
import { Dividers } from './Dividers'

export class BoardManager {
  reader: BoardReader
  watcher: BoardWatcher
  board3D: Custom3DBoard | null = null
  dividers: Dividers
  private cleanupEffect: () => void

  constructor() {
    this.reader = new BoardReader()
    this.watcher = new BoardWatcher()
    this.dividers = new Dividers()
  }

  init() {
    this.watcher.start()

    // React to board changes - update 3D board
    this.watcher.onChange(() => {
      if (this.board3D) {
        const pieces = this.reader.readPiecePositions()
        this.board3D.render(pieces)
      }
    })

    // React to settings changes - reapply everything
    this.cleanupEffect = effect(() => {
      this.reapplyAll()
    })
  }

  private reapplyAll() {
    // Custom board
    if (settings.customBoardEnabled.value) {
      if (!this.board3D) {
        this.board3D = new Custom3DBoard()
        this.board3D.init()
      }
      this.board3D.updateParallax(settings.parallax.value)
      this.board3D.updateHoverMode(settings.hoverMode.value)

      if (settings.obfuscationsEnabled.value) {
        this.board3D.updatePieceStyle(settings.pieceStyle.value)
        this.board3D.updateBlur(settings.blur.value)
        this.board3D.updateBlackSegments(
          settings.blackSegments.value,
          settings.blackSegmentsTiming.value
        )
      }
    } else {
      // Clean up 3D board
      if (this.board3D) {
        this.board3D.destroy()
        this.board3D = null
      }
    }

    // Dividers
    if (settings.dividersEnabled.value) {
      const is3DMode = settings.customBoardEnabled.value
      this.dividers.render(is3DMode)
    } else {
      this.dividers.hide()
    }
  }

  destroy() {
    this.watcher.stop()
    this.board3D?.destroy()
    this.dividers.destroy()
    this.cleanupEffect() // Stop watching signals
  }
}

FlashManager with Signals

flash/FlashManager.ts
import { effect } from '@preact/signals-core'
import { settings } from '../settings/settingsStore'
import { FlashOverlay } from './FlashOverlay'
import { boardWatcher } from '../board/BoardWatcher'

export class FlashManager {
  private overlay: FlashOverlay
  private durationTimer: number | null = null
  private intervalTimer: number | null = null
  private cleanupEffect: () => void
  private cleanupWatcher: () => void

  constructor() {
    this.overlay = new FlashOverlay()
  }

  init() {
    // React to board changes
    this.cleanupWatcher = boardWatcher.onChange(() => {
      if (settings.flashModeEnabled.value) {
        this.onBoardChange()
      }
    })

    // React to settings changes
    this.cleanupEffect = effect(() => {
      this.reapply()
    })
  }

  private reapply() {
    if (settings.flashModeEnabled.value) {
      this.overlay.show()
      this.startIntervalFlash()
    } else {
      this.overlay.hide()
      this.stopIntervalFlash()
    }
  }

  private onBoardChange() {
    // Show board briefly
    this.overlay.hide()

    // Clear existing timer
    if (this.durationTimer) {
      clearTimeout(this.durationTimer)
    }

    // Hide after duration
    const duration = settings.flashDuration.value * 1000
    this.durationTimer = setTimeout(() => {
      this.overlay.show()
    }, duration)

    // Reset interval timer
    this.stopIntervalFlash()
    this.startIntervalFlash()
  }

  private startIntervalFlash() {
    const interval = settings.flashInterval.value * 1000
    this.intervalTimer = setInterval(() => {
      this.onBoardChange()
    }, interval)
  }

  private stopIntervalFlash() {
    if (this.intervalTimer) {
      clearInterval(this.intervalTimer)
      this.intervalTimer = null
    }
  }

  destroy() {
    if (this.durationTimer) clearTimeout(this.durationTimer)
    this.stopIntervalFlash()
    this.overlay.destroy()
    this.cleanupEffect()
    this.cleanupWatcher()
  }
}

SpeechManager with Signals

speech/SpeechManager.ts
import { effect } from '@preact/signals-core'
import { settings } from '../settings/settingsStore'
import { SpeechSynthesizer } from './SpeechSynthesizer'
import { boardReader } from '../board/BoardReader'
import * as textGen from './textGenerators'

export class SpeechManager {
  private synthesizer: SpeechSynthesizer
  private cleanupEffect: () => void

  constructor() {
    this.synthesizer = new SpeechSynthesizer()
  }

  init() {
    // React to speak rate changes
    this.cleanupEffect = effect(() => {
      this.synthesizer.setRate(settings.speakRate.value)
    })
  }

  speakQuadrant(quadrant: 'wk' | 'wq' | 'bk' | 'bq') {
    const pieces = boardReader.readPiecePositions()
    const text = textGen.generateQuadrantText(pieces, quadrant)
    this.synthesizer.speak(text)
  }

  speakAll() {
    const pieces = boardReader.readPiecePositions()
    const text = textGen.generateAllPiecesText(pieces)
    this.synthesizer.speak(text)
  }

  speakColor(color: 'white' | 'black') {
    const pieces = boardReader.readPiecePositions()
    const text = textGen.generateColorText(pieces, color)
    this.synthesizer.speak(text)
  }

  stop() {
    this.synthesizer.stop()
  }

  destroy() {
    this.cleanupEffect()
  }
}

Main Entry Point (Revised)

main.ts
import { render } from 'preact'
import { loadSettings } from './settings/settingsStore'
import { ControlPanel } from './components/ControlPanel'
import { BoardManager } from './board/BoardManager'
import { SpeechManager } from './speech/SpeechManager'
import { DrawingManager } from './drawing/DrawingManager'
import { FlashManager } from './flash/FlashManager'
import { setupCommandHandler } from './commands/commandHandler'

async function waitForElement(selector: string) {
  // ... implementation
}

async function init() {
  // Wait for lichess DOM
  await waitForElement('.keyboard-move')

  // Load settings from localStorage
  loadSettings()

  // Initialize managers
  const boardManager = new BoardManager()
  const speechManager = new SpeechManager()
  const drawingManager = new DrawingManager()
  const flashManager = new FlashManager()

  boardManager.init()
  speechManager.init()
  flashManager.init()

  // Set up keyboard commands
  setupCommandHandler(speechManager, drawingManager)

  // Mount Preact UI
  const mountPoint = document.createElement('div')
  document.querySelector('.keyboard-move')?.appendChild(mountPoint)
  render(<ControlPanel />, mountPoint)

  // Return cleanup function for page navigation
  return () => {
    boardManager.destroy()
    speechManager.destroy()
    drawingManager.destroy()
    flashManager.destroy()
    render(null, mountPoint)
  }
}

Key differences from callback pattern:

  • Managers use effect(() => { ... }) instead of onChange(callback)
  • effect() automatically tracks which signals are read inside it
  • When any tracked signal changes, effect runs again
  • Cleanup functions returned from effect() for proper teardown
  • Simpler code - no manual listener arrays or notifyListeners()
