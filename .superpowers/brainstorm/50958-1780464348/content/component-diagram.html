Settings Module

settings/SettingsStore.ts
class SettingsStore extends EventEmitter {
  // All 14 settings as properties
  speakRate: number
  piecesListEnabled: boolean
  dividersEnabled: boolean
  customBoardEnabled: boolean
  obfuscationsEnabled: boolean
  parallax: number
  hoverMode: string
  pieceStyle: string
  blur: number
  blackSegments: string
  blackSegmentsTiming: string
  flashModeEnabled: boolean
  flashDuration: number
  flashInterval: number

  update(key, value) // Updates setting, saves, emits event
  get(key) // Returns current value
  getAll() // Returns all settings
  load() // Loads from localStorage
  save() // Saves to localStorage
}

Board Module

board/BoardReader.ts
class BoardReader {
  readPiecePositions() // Parses lichess DOM

board/BoardWatcher.ts
class BoardWatcher extends EventEmitter {
  start() // Sets up MutationObserver on cg-board
  stop() // Disconnects observer
  // Emits 'board:changed' when pieces move
}

board/Custom3DBoard.ts
class Custom3DBoard {
  canvas: HTMLCanvasElement
  scene: THREE.Scene
  camera: THREE.Camera
  renderer: THREE.Renderer

  init() // Creates Three.js scene
  reapply(settings) // Updates parallax, hover, piece style, blur
  render(pieces) // Renders current piece positions
  destroy() // Cleans up canvas/scene
}

board/Dividers.ts
class Dividers {
  svg: SVGElement
  reapply(settings) // Show/hide dividers, 2D vs 3D mode
  render() // Draws quadrant lines
  destroy() // Removes SVG
}

board/BoardManager.ts (Coordinator)
class BoardManager {
  reader: BoardReader
  watcher: BoardWatcher
  board3D: Custom3DBoard
  dividers: Dividers

  init() // Sets up all board subsystems
  reapply(settings) // Reapplies custom board, obfuscations, dividers
  destroy() // Cleanup
}

Main Entry Point

main.ts
// Wait for lichess DOM
waitForElement('.keyboard-move').then(() => {
  // Initialize systems
  const settingsStore = new SettingsStore()
  settingsStore.load()

  const boardManager = new BoardManager(settingsStore)
  const speechManager = new SpeechManager(settingsStore, boardManager.reader)
  const drawingManager = new DrawingManager()
  const flashManager = new FlashManager(settingsStore)

  boardManager.init()

  // Wire up settings changes
  settingsStore.on('settings:changed', () => {
    boardManager.reapply(settingsStore.getAll())
    speechManager.reapply(settingsStore.getAll())
    flashManager.reapply(settingsStore.getAll())
  })

  // Wire up board changes to flash mode
  boardManager.watcher.on('board:changed', () => {
    flashManager.onBoardChange()
    // Vue PiecesList also listens to this
  })

  // Mount Vue UI
  const app = createApp(ControlPanel)
  app.provide('settingsStore', settingsStore)
  app.provide('boardReader', boardManager.reader)
  app.provide('speechManager', speechManager)
  app.provide('drawingManager', drawingManager)
  app.mount(mountPoint)

  // Initial apply
  settingsStore.emit('settings:changed')

  // Set up keyboard command handler
  setupCommandHandler(speechManager, drawingManager)
})
