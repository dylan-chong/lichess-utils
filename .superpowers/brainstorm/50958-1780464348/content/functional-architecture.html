const STORAGE_KEY = 'lichess-board-speaker-settings'

export function loadSettings() {
  const stored = localStorage.getItem(STORAGE_KEY)
  if (!stored) return

  const data = JSON.parse(stored)
  Object.keys(data).forEach(key => {
    if (settings[key]) {
      settings[key].value = data[key]
    }
  })
}

export function saveSettings() {
  const data = {}
  Object.keys(settings).forEach(key => {
    data[key] = settings[key].value
  })
  localStorage.setItem(STORAGE_KEY, JSON.stringify(data))
}

// Auto-save on any change
effect(() => {
  Object.values(settings).forEach(s => s.value)
  saveSettings()
})

BoardReader (Pure Functions)

board/boardReader.ts
interface Piece {
  color: 'white' | 'black'
  type: string
  square: string
}

export function getPlayerColor(): 'white' | 'black' {
  const coords = document.querySelector('coords')
  return coords?.classList.contains('black') ? 'black' : 'white'
}

export function readPiecePositions(): Piece[] {
  const pieces: Piece[] = []
  const board = document.querySelector('cg-board:not(.userscript-custom-board)')
  if (!board) return pieces

  const squareSize = board.offsetWidth / 8
  const isPlayingWhite = getPlayerColor() === 'white'

  board.querySelectorAll('piece').forEach(piece => {
    const classes = piece.className.split(' ')
    const color = classes[0] as 'white' | 'black'
    const type = classes[1]

    const transform = piece.style.transform
    const match = transform.match(/translate\((.+?)px(?:,\s*(.+?)px)?\)/)
    if (!match) return

    const x = parseFloat(match[1])
    const y = parseFloat(match[2] || match[1])

    const square = pixelsToSquare(x, y, squareSize, isPlayingWhite)
    pieces.push({ color, type, square })
  })

  return pieces
}

function pixelsToSquare(
  x: number,
  y: number,
  squareSize: number,
  isPlayingWhite: boolean
): string {
  let col = Math.round(x / squareSize)
  let row = Math.round(y / squareSize)

  if (isPlayingWhite) {
    row = 7 - row
  } else {
    col = 7 - col
  }

  const file = 'abcdefgh'[col]
  const rank = row + 1
  return `${file}${rank}`
}

BoardWatcher (Factory Function)

board/boardWatcher.ts
export function createBoardWatcher() {
  let observer: MutationObserver | null = null
  let listeners: Array<() => void> = []

  function notifyListeners() {
    listeners.forEach(cb => cb())
  }

  function onChange(callback: () => void): () => void {
    listeners.push(callback)
    return () => {
      listeners = listeners.filter(cb => cb !== callback)
    }
  }

  function start() {
    const board = document.querySelector('cg-board')
    if (!board) return

    observer = new MutationObserver(notifyListeners)
    observer.observe(board, {
      childList: true,
      attributes: true,
      subtree: true,
    })
  }

  function stop() {
    observer?.disconnect()
    observer = null
    listeners = []
  }

  return { start, stop, onChange }
}

export function createDividers() {
  let svg: SVGElement | null = null

  function render(is3DMode: boolean) {
    if (!svg) {
      svg = document.createElementNS('http://www.w3.org/2000/svg', 'svg')
      svg.classList.add('userscript-dividers')
      // ... setup
    }

    const parent = is3DMode
      ? document.querySelector('cg-container')
      : document.querySelector('cg-board')

    parent?.appendChild(svg)
    svg.style.display = 'block'
    // Draw lines...
  }

  function hide() {
    if (svg) svg.style.display = 'none'
  }

  function destroy() {
    svg?.remove()
    svg = null
  }

  return { render, hide, destroy }
}

Pattern: Factory functions return objects with public methods. Internal state lives in closure scope. No this, no class, no new.
